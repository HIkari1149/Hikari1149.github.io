2-3 编写一个add()函数,能正常执行下列代码.

实参数量不定,使用类数组对象arguments.
add(1)(2)(3),在函数内部在定义一个函数,该内部函数每次合并传入的实参,并返回自身.
最后通过重写toString方法(求和).在console.log()时就会调用toString方法.

function add()
           {
               var tmpSlice=[].slice,params=tmpSlice.apply(arguments);//间接调用slice()方法
               function currying()
               {
                   var arr=tmpSlice.apply(arguments);
                   //由于闭包的关系,所有能读取params变量.
                   params=params.concat(arr);//合并参数
                   return currying;
               }
               currying.toString=function ()
               {
                   var res=0;
                   params.forEach(function (value) {
                      res+=value;
                   });
                   return res;
               }
               return currying;
           }
